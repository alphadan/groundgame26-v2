1. Update the Meta Table

CREATE OR REPLACE TABLE `groundgame26-v2.groundgame26_voters.meta_20260225_chester_county` AS
SELECT 
  precinct,
  area_district,
  -- Turnout & Recovery
  COUNTIF(political_party = 'R' AND GN_11_05_24 = 'R') AS gop_ballots_24,
  COUNTIF(political_party = 'R' AND GN_PR_11_04_25 = 'R') AS gop_ballots_25_primary,
  (COUNTIF(political_party = 'R' AND GN_11_05_24 = 'R') - COUNTIF(political_party = 'R' AND GN_PR_11_04_25 = 'R')) AS gop_falloff_count,
  
  -- The Score (Ensure this name matches exactly)
  AVG(turnout_score_general) AS avg_rep_gen_score,

  -- VBM & Demographics
  COUNTIF(vbm_ballotsent IS NOT NULL) AS vbm_sent_count,
  COUNTIF(vbm_ballotreturned IS NOT NULL) AS vbm_returned_count,
  ROUND(AVG(IF(age > 65, 1, 0)) * 100, 1) AS pct_seniors,
  IF(precinct IN ('425'), 1, 0) AS is_known_college_precinct

FROM `groundgame26-v2.groundgame26_voters.20260225_chester_county`
GROUP BY 1, 2;

2. Create Gold Stadard Table

CREATE OR REPLACE TABLE `groundgame26-v2.groundgame26_voters.gold_standard_feb2026` AS
SELECT 
  m.*, -- This ensures avg_rep_gen_score is passed through
  v24.rep_candidate_total AS rep_24,
  v24.dem_candidate_total AS dem_24,
  COALESCE(SAFE_CAST(intel.ai_adjustment_flag AS INT64), 0) AS human_intel_flag
FROM `groundgame26-v2.groundgame26_voters.meta_20260225_chester_county` m
JOIN `groundgame26-v2.groundgame26_voters.baseline_2024_general` v24 ON m.precinct = v24.precinct_code
JOIN `groundgame26-v2.groundgame26_voters.baseline_2022_general` v22 ON m.precinct = v22.precinct_code
LEFT JOIN `groundgame26-v2.groundgame26_voters.human_intel_context` intel ON m.precinct = intel.precinct_code;

3. Update the Volunteer Quotas Table

CREATE OR REPLACE TABLE `groundgame26-v2.groundgame26_voters.precinct_field_plan_2026` AS
WITH Benchmarks AS (
  SELECT MAX(avg_rep_gen_score) as top_gen_score
  FROM `groundgame26-v2.groundgame26_voters.gold_standard_feb2026`
),
-- Step 1: Calculate all raw numbers first
CalculatedMetrics AS (
  SELECT 
    m.*,
    CAST(CEIL(m.gop_falloff_count * 0.10) AS INT64) AS goal_new_regs,
    CAST(CEIL(m.rep_24 * 0.20) AS INT64) AS goal_vbm_signups,
    CAST(CEIL((b.top_gen_score - m.avg_rep_gen_score) * 10 + (m.gop_falloff_count * 0.5)) AS INT64) AS goal_total_knocks,
    CAST(CEIL(((b.top_gen_score - m.avg_rep_gen_score) * 10 + (m.gop_falloff_count * 0.5)) / 40) AS INT64) AS volunteers_needed_quota
  FROM `groundgame26-v2.groundgame26_voters.gold_standard_feb2026` m, Benchmarks b
)
-- Step 2: Generate the 4-tier Verbose Narratives using the calculated metrics
SELECT 
  precinct,
  area_district,
  goal_new_regs,
  goal_vbm_signups,
  goal_total_knocks,
  volunteers_needed_quota,
  0 AS actual_knocks_to_date,
  0 AS actual_regs_to_date,

  -- 1. EXECUTIVE SUMMARY
  CASE 
    WHEN is_known_college_precinct = 1 THEN 'High-transience campus environment (425) requiring a focus on registration and leadership recruitment.'
    WHEN gop_falloff_count > 150 THEN 'Critical recovery zone. High-density of 2024 voters who sat out the 2025 primary.'
    ELSE 'Stable residential sector. Strategy shifts toward VBM conversion and surplus volunteer harvesting.'
  END AS verbose_strategic_narrative,

  -- 2. POSITIVE TRENDS
  CASE 
    WHEN avg_rep_gen_score > 8.0 THEN 'Top-tier reliability. This precinct serves as a performance benchmark for the district.'
    WHEN (vbm_sent_count - vbm_returned_count) < 10 THEN 'High VBM efficiency. Ballot return rates are currently outpacing the county average.'
    ELSE 'Solid foundation. Historical data shows a resilient Republican floor.'
  END AS verbose_strategic_positive,

  -- 3. IMMEDIATE ATTENTION
  CASE 
    WHEN (vbm_sent_count - vbm_returned_count) > 50 THEN 'IMMEDIATE: VBM Leakage. Over 50 ballots remain unreturned, representing a significant risk.'
    WHEN is_known_college_precinct = 1 THEN 'IMMEDIATE: Roll Cleanliness. Address verification required for 18-24 demographic.'
    ELSE 'IMMEDIATE: Monitor registration pace to ensure 10% monthly growth.'
  END AS verbose_strategic_immediate,

  -- 4. ACTIONABLE STEPS
  CASE 
    WHEN is_known_college_precinct = 1 THEN 'Action: Deploy 5 volunteers to high-traffic student zones for registration drives.'
    WHEN gop_falloff_count > 100 THEN 'Action: Execute "Recovery Script" at doors. Focus on the ' || CAST(gop_falloff_count AS STRING) || ' missing 2025 voters.'
    ELSE 'Action: Pivot ' || CAST(volunteers_needed_quota AS STRING) || ' volunteers to support nearest high-falloff zone.'
  END AS verbose_strategic_actionable

FROM CalculatedMetrics;

5. Copy to Firebase colletion with gcloud 

from google.cloud import bigquery
from google.cloud import firestore

def deploy_narratives():
    # Initialize clients
    bq_client = bigquery.Client()
    db = firestore.Client()

    # Define the source table
    table_id = "groundgame26-v2.groundgame26_voters.strategic_narratives_feb2026"
    
    # Query the table
    query = f"SELECT * FROM `{table_id}`"
    results = bq_client.query(query).result()

    print(f"Starting deployment to Firestore...")

    # Write each precinct to Firestore
    for row in results:
        doc_id = f"precinct_{row.precinct}"
        doc_ref = db.collection("strategic_narrative").document(doc_id)
        
        # Prepare the data object
        data = {
            "area_district": row.area_district,
            "precinct": row.precinct,
            "area_chair_narrative": row.area_chair_narrative,
            "precinct_narrative": row.precinct_narrative,
            "total_volunteers_quota": row.total_volunteers_quota,
            "targets": {
                "doors": row.targets['doors_needed'],
                "sms": row.targets['sms_needed']
            }
        }
        
        doc_ref.set(data)
        print(f"Deployed narrative for {doc_id}")

    print("Deployment complete.")

if __name__ == "__main__":
    deploy_narratives()

CREATE OR REPLACE TABLE `groundgame26-v2.groundgame26_voters.firebase_flattened_precincts` AS
SELECT
  -- Extract Precinct ID from event parameters
  (SELECT value.string_value FROM UNNEST(event_params) WHERE key = 'precinct_id') AS precinct,
  -- Aggregate interactions
  COUNTIF(event_name = 'door_knock') AS total_doors,
  COUNTIF(event_name = 'send_email') AS total_emails,
  COUNTIF(event_name = 'phone_call') AS total_calls,
  COUNTIF(event_name = 'sms_text') AS total_sms,
  -- Measure engagement
  AVG( (SELECT value.int_value FROM UNNEST(event_params) WHERE key = 'engagement_time_msec') / 1000 ) AS avg_session_seconds
FROM
  `your-project.analytics_12345.events_*`
GROUP BY 1;

9. Prepare Survey Results for ML

CREATE OR REPLACE TABLE `groundgame26-v2.groundgame26_voters.enriched_strategy_feb2026` AS
SELECT
  m.*,
  s.avg_issue_score,
  s.top_concern
FROM `groundgame26-v2.groundgame26_voters.meta_20260225_chester_county` m
LEFT JOIN (
  -- Aggregate survey results by cohort to match the Meta Table
  SELECT 
    age_cohort, 
    political_party, 
    sex, 
    AVG(issue_score) as avg_issue_score,
    APPROX_TOP_SUM(concern, 1, 1)[OFFSET(0)].value as top_concern
  FROM `groundgame26-v2.groundgame26_voters.survey_results_feb2026`
  GROUP BY 1, 2, 3
) s 
ON m.age_group = s.age_cohort 
AND m.political_party = s.political_party 
AND m.sex = s.sex;

10. The "Executive Summary" Logic Table

CREATE OR REPLACE VIEW `groundgame26-v2.groundgame26_voters.executive_dashboard_view` AS
WITH Metrics AS (
  SELECT
    SUM(current_rep_count) as total_reps,
    SUM(total_volunteers_quota) as quota_volunteers,
    -- Join with Firebase to see actuals
    (SELECT COUNT(DISTINCT user_id) FROM `groundgame26-v2.groundgame26_voters.firebase_flattened_precincts`) as actual_volunteers,
    (SELECT SUM(total_doors) FROM `groundgame26-v2.groundgame26_voters.firebase_flattened_precincts`) as actual_doors
  FROM `groundgame26-v2.groundgame26_voters.volunteer_quotas_feb2026`
)
SELECT
  SAFE_DIVIDE(actual_volunteers, quota_volunteers) as pct_vol_goal,
  SAFE_DIVIDE(actual_doors, (quota_volunteers * 50)) as pct_contact_goal,
  '2.1%' as reg_growth_rate -- This would be a calculation vs. previous month
FROM Metrics;
