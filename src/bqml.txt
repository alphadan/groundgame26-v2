1. Create Meta Table Snapshot meta_20260225_chester_county

CREATE OR REPLACE TABLE `groundgame26-v2.groundgame26_voters.meta_20260225_chester_county` AS
SELECT 
  precinct,
  area_district,
  
  -- Original Turnout Logic
  COUNTIF(political_party = 'R' AND GN_11_05_24 = 'R') AS gop_ballots_24,
  COUNTIF(political_party = 'R' AND GN_PR_11_04_25 = 'R') AS gop_ballots_25_primary,
  (COUNTIF(political_party = 'R' AND GN_11_05_24 = 'R') - COUNTIF(political_party = 'R' AND GN_PR_11_04_25 = 'R')) AS gop_falloff_count,
  
  -- Original Scores & Flags
  AVG(turnout_score_general) AS avg_rep_gen_score,
  COUNTIF(vbm_ballotsent IS NOT NULL) AS vbm_sent_count,
  COUNTIF(vbm_ballotreturned IS NOT NULL) AS vbm_returned_count,
  ROUND(AVG(IF(age > 65, 1, 0)) * 100, 1) AS pct_seniors,
  IF(precinct IN ('425'), 1, 0) AS is_known_college_precinct,

  -- NEW: Demographic Pivots for AI Narratives
  COUNTIF(age BETWEEN 18 AND 24) AS age_18_24,
  COUNTIF(age BETWEEN 25 AND 40) AS age_25_40,
  COUNTIF(age BETWEEN 41 AND 70) AS age_41_70,
  COUNTIF(age >= 71) AS age_71_plus,
  COUNTIF(sex = 'M') AS male_count,
  COUNTIF(sex = 'F') AS female_count

FROM `groundgame26-v2.groundgame26_voters.20260225_chester_county`
GROUP BY 1, 2;

2. The "Gold Standard" Table (Momentum Logic)

CREATE OR REPLACE TABLE `groundgame26-v2.groundgame26_voters.gold_standard_poc` AS
SELECT 
  m.*,
  -- 2024 Baseline Stats
  v24.rep_candidate_total AS rep_24,
  v24.ballots_cast AS total_24,
  -- 2022 Baseline Stats
  v22.rep_candidate_total AS rep_22,
  v22.ballots_cast AS total_22,
  -- MOMENTUM: (GOP 2024 Share) - (GOP 2022 Share)
  (SAFE_DIVIDE(v24.rep_candidate_total, v24.ballots_cast) - 
   SAFE_DIVIDE(v22.rep_candidate_total, v22.ballots_cast)) AS gop_momentum
FROM `groundgame26-v2.groundgame26_voters.meta_20260225_chester_county` m
LEFT JOIN `groundgame26-v2.groundgame26_voters.baseline_2024_general` v24 ON m.precinct = v24.precinct_code
LEFT JOIN `groundgame26-v2.groundgame26_voters.baseline_2022_general` v22 ON m.precinct = v22.precinct_code;

3. Weighted Quota Engine for precinct_quotas_2026

CREATE OR REPLACE TABLE `groundgame26-v2.groundgame26_voters.precinct_quotas_2026` AS
WITH Benchmarks AS (
  SELECT MAX(avg_rep_gen_score) as top_gen_score
  FROM `groundgame26-v2.groundgame26_voters.gold_standard_poc`
)
SELECT 
  m.precinct,
  m.area_district,
  
  -- 1. WEIGHTED REGISTRATION GOALS
  -- We increase the % goal if GOP Momentum is positive (striking while the iron is hot)
  CAST(CEIL(m.gop_falloff_count * (0.10 + IF(m.gop_momentum > 0, 0.05, 0))) AS INT64) AS goal_new_regs,
  
  -- 2. TARGETED VBM GOALS
  -- If Seniors > 40%, we push for 30% conversion. Otherwise, stick to 15%.
  CAST(CEIL(m.gop_ballots_24 * IF(m.pct_seniors > 40, 0.30, 0.15)) AS INT64) AS goal_vbm_signups,
  
  -- 3. INTERACTION QUOTAS (Knocks & SMS)
  -- Focused on the 'Efficiency Gap' from the top performing precinct
  CAST(CEIL((b.top_gen_score - m.avg_rep_gen_score) * 10 + (m.gop_falloff_count * 0.4)) AS INT64) AS goal_total_knocks,
  CAST(CEIL((b.top_gen_score - m.avg_rep_gen_score) * 25 + (m.gop_falloff_count * 0.8)) AS INT64) AS goal_total_sms,

  -- 4. INITIAL ACTUALS (Reset for PoC)
  0 AS actual_regs_to_date,
  0 AS actual_knocks_to_date,
  0 AS actual_sms_to_date,
  0 AS actual_vbm_to_date,

  -- 5. REFINED STRATEGIC PRIORITY
  -- Using your new demographic pivots for better classification
  CASE 
    WHEN m.is_known_college_precinct = 1 THEN 'YOUTH_REG_FOCUS'
    WHEN m.pct_seniors > 40 AND m.female_count > m.male_count THEN 'SENIOR_VBM_CHASE'
    WHEN m.avg_rep_gen_score > 2.5 THEN 'BASE_MAXIMIZATION'
    ELSE 'SWING_PERSUASION'
  END AS strategic_priority

FROM `groundgame26-v2.groundgame26_voters.gold_standard_poc` m, Benchmarks b;

4. The AI Narrative Strategic Query precinct_field_plan_2026

CREATE OR REPLACE TABLE `groundgame26-v2.groundgame26_voters.precinct_field_plan_2026` AS
WITH AI_Capture AS (
  SELECT *
  FROM ML.GENERATE_TEXT(
    MODEL `groundgame26-v2.groundgame26_voters.gemini_flash_model`,
    (
      SELECT 
        precinct,
        area_district,
        gop_falloff_count,
        pct_seniors,
        is_known_college_precinct,
        -- Injecting the deep context we built
        CONCAT(
          "ACT AS A GOP FIELD DIRECTOR. Objective: 15k voter recovery. ",
          "Analyze Precinct ", precinct, ". Falloff: ", gop_falloff_count, ". ",
          "Seniors: ", pct_seniors, "%. College: ", IF(is_known_college_precinct=1, 'Yes', 'No'), ". ",
          "STRICT JSON ONLY. NO MARKDOWN. ",
          "Format: {'summary': '...', 'positive': '...', 'immediate': '...', 'actionable': '...'}"
        ) AS prompt
      FROM `groundgame26-v2.groundgame26_voters.gold_standard_poc`
    ),
    STRUCT(0.1 AS temperature, 1000 AS max_output_tokens, TRUE AS flatten_json_output)
  )
),
Sanitized AS (
  SELECT
    precinct,
    area_district,
    gop_falloff_count,
    pct_seniors,
    is_known_college_precinct,
    -- Your Regex Cleanup logic
    REGEXP_REPLACE(ml_generate_text_llm_result, r'^```json|```$', '') as json_blob
  FROM AI_Capture
),
Parsed AS (
  SELECT
    *,
    SAFE.PARSE_JSON(json_blob) as json_obj
  FROM Sanitized
)
SELECT
  precinct,
  area_district,
  -- AI with Human Fallback
  COALESCE(JSON_VALUE(json_obj, '$.summary'), 'Focusing on recovery of ' || gop_falloff_count || ' falloff voters.') as verbose_strategic_summary,
  COALESCE(JSON_VALUE(json_obj, '$.positive'), 'Strong GOP historical floor identified.') as verbose_strategic_positive,
  COALESCE(JSON_VALUE(json_obj, '$.immediate'), 
    IF(pct_seniors > 40, 'IMMEDIATE: High senior density - VBM chase required.', 'IMMEDIATE: Door-to-door verification.')
  ) as verbose_strategic_immediate,
  COALESCE(JSON_VALUE(json_obj, '$.actionable'), 
    IF(is_known_college_precinct = 1, 'Action: Student registration drive.', 'Action: SMS/Phone outreach to ' || gop_falloff_count || ' voters.')
  ) as verbose_strategic_actionable,
  UNIX_SECONDS(CURRENT_TIMESTAMP()) as updated_at
FROM Parsed;
