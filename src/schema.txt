8. Calculated age_group field

UPDATE `groundgame26-v2.groundgame26_voters.20260225_chester_county`
SET age_group = CASE 
    WHEN age BETWEEN 18 AND 25 THEN '18-25'
    WHEN age > 25 AND age <= 40 THEN '26-40'
    WHEN age >= 41 AND age <= 70 THEN '41-70'
    WHEN age > 70 THEN '71+'
    ELSE 'Unknown'
END
WHERE age IS NOT NULL;
